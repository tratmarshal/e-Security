// ===== GET SUBSTITUTE LIST (รายชื่อผู้แทนเวร) =====

function getSubstituteList() {
  try {
    var rows = SheetService.getValues(SHEETS.SUPLIST, 'A:C');
    if (rows.length < 2) return jsonResponse(true, 'ไม่มีข้อมูล', []);

    var list = [];
    for (var i = 1; i < rows.length; i++) {
      if (rows[i][COL.SUPLIST.NAME]) {
        list.push({
          name: rows[i][COL.SUPLIST.NAME] || '',
          employeeId: rows[i][COL.SUPLIST.EMPLOYEE_ID] || ''
        });
      }
    }

    // เรียงตามชื่อ ก-ฮ
    list.sort(function(a, b) {
      return a.name.localeCompare(b.name, 'th');
    });

    return jsonResponse(true, '', list);
  } catch (err) {
    return handleError(err);
  }
}

// ===== SUBMIT SWAP (บันทึกการลา) — รองรับลาห้วงเวลา + ป้องกันผู้แทนซ้ำ + ตรวจสอบตาม Business Rules =====

function submitSwap(payload) {
  var lock = LockService.getScriptLock();
  lock.waitLock(30000);
  try {
    // 1. Validate required fields
    var validation = validatePayload(payload, ['lineUserId', 'swapDateStart', 'swapDateEnd', 'shift', 'substituteEmployeeId']);
    if (!validation.valid) return jsonResponse(false, validation.message);

    // 2. Validate dates
    var dateStartValidation = validateDate(payload.swapDateStart);
    if (!dateStartValidation.valid) return jsonResponse(false, dateStartValidation.message);
    var dateEndValidation = validateDate(payload.swapDateEnd);
    if (!dateEndValidation.valid) return jsonResponse(false, dateEndValidation.message);

    if (payload.swapDateEnd < payload.swapDateStart) {
      return jsonResponse(false, 'วันที่สิ้นสุดต้องไม่ก่อนวันที่เริ่ม');
    }

    // 3. Anti-spam check
    var spamCheck = checkSpam(payload.lineUserId, 'submitSwap');
    if (!spamCheck.valid) return jsonResponse(false, spamCheck.message);

    // 4. Authenticate (ตรวจสอบสิทธิ์) — ได้ชื่อผู้ลาและ employeeId
    var auth = authenticateRequest(payload);
    if (!auth.success) {
      logAction('submitSwap', payload.lineUserId, 'ลา', 'FAIL', auth.message);
      return jsonResponse(false, auth.message);
    }

    // 4. ห้ามเลือกตัวเองเป็นผู้แทนเวร (เทียบ employeeId)
    if (auth.data.employeeId && auth.data.employeeId === payload.substituteEmployeeId) {
      logAction('submitSwap', payload.lineUserId, 'ลา', 'FAIL', 'เลือกตัวเองเป็นผู้แทน');
      return jsonResponse(false, 'ไม่สามารถเลือกตนเองเป็นผู้แทนเวรได้');
    }

    // เรียกชีทออกมาก่อนเพื่อนำมาประมวลผลอย่างปลอดภัย
    var sheet = SheetService.getSheet(SHEETS.SWAP);
    var ssId = sheet.getParent().getId();
    var sheetId = sheet.getSheetId();

    // 5. ดึงชื่อผู้แทน (จากชีท SUPLIST)
    var substituteName = '';
    var listRows = SheetService.getValues(SHEETS.SUPLIST, 'A:C');
    
    for (var j = 1; j < listRows.length; j++) {
      if (listRows[j][COL.SUPLIST.EMPLOYEE_ID] && String(listRows[j][COL.SUPLIST.EMPLOYEE_ID]).trim() === payload.substituteEmployeeId) {
        substituteName = listRows[j][COL.SUPLIST.NAME] || '';
        break;
      }
    }

    // 6. สร้างรายการวันที่ทั้งหมดในช่วง
    var dates = [];
    var current = new Date(payload.swapDateStart);
    var end = new Date(payload.swapDateEnd);
    while (current <= end) {
      var y = current.getFullYear();
      var m = String(current.getMonth() + 1).padStart(2, '0');
      var d = String(current.getDate()).padStart(2, '0');
      dates.push(y + '-' + m + '-' + d);
      current.setDate(current.getDate() + 1);
    }

    // ==========================================
    // UNIFIED VALIDATION BLOCK (Rules 1, 2, 3)
    // ==========================================

    // อ่านข้อมูลจากชีท SWAP เพียงรอบเดียวสำหรับทำ Validation ทุกข้อ
    var existingRows = SheetService.getValues(SHEETS.SWAP, 'A:G');

    var colSubEmployeeId = (typeof COL !== 'undefined' && COL.SWAP && COL.SWAP.SUBLIST_EMPLOYEE_ID !== undefined) 
      ? COL.SWAP.SUBLIST_EMPLOYEE_ID 
      : 4;

    var reqLineUserId = String(payload.lineUserId).trim();
    var reqEmpId = String(auth.data.employeeId || '').trim();
    var subEmpId = String(payload.substituteEmployeeId).trim();
    var proposedShift = String(payload.shift).trim();

    var deleteIndices = [];
    var activeSubstitutes = {}; // เก็บสถานะปฏิบัติงานจริง { employeeId: { "yyyy-MM-dd|shift": true } }

    // ฟังก์ชันช่วยจัดรูปแบบวันที่เป็น yyyy-MM-dd
    function parseRowDate(cellValue) {
      if (!cellValue) return "";
      if (cellValue instanceof Date) {
        return Utilities.formatDate(cellValue, Session.getScriptTimeZone(), 'yyyy-MM-dd');
      }
      var tempDate = new Date(cellValue);
      if (!isNaN(tempDate.getTime())) {
        return Utilities.formatDate(tempDate, Session.getScriptTimeZone(), 'yyyy-MM-dd');
      }
      return String(cellValue).trim();
    }

    // ฟังก์ชันคำนวณวันข้างเคียง (บวก/ลบวัน)
    function getAdjacentDateStr(dateStr, offsetDays) {
      var parts = dateStr.split('-');
      var d = new Date(parseInt(parts[0]), parseInt(parts[1]) - 1, parseInt(parts[2]));
      d.setDate(d.getDate() + offsetDays);
      var y = d.getFullYear();
      var m = String(d.getMonth() + 1).padStart(2, '0');
      var day = String(d.getDate()).padStart(2, '0');
      return y + '-' + m + '-' + day;
    }

    // ฟังก์ชันตรวจการควงกะ (ห้าม กลางวัน X -> กลางคืน X และ ห้าม กลางคืน X -> กลางวัน X+1)
    function hasConsecutiveShifts(scheduleMap) {
      for (var key in scheduleMap) {
        if (!scheduleMap[key]) continue;
        var parts = key.split('|');
        var dStr = parts[0];
        var shift = parts[1];

        if (shift === "กลางวัน") {
          // ตรวจผลัดกลางคืนในวันเดียวกัน
          if (scheduleMap[dStr + '|กลางคืน']) return true;
        } else if (shift === "กลางคืน") {
          // ตรวจผลัดกลางวันในวันถัดไป
          var nextDayStr = getAdjacentDateStr(dStr, 1);
          if (scheduleMap[nextDayStr + '|กลางวัน']) return true;
        }
      }
      return false;
    }

    // ขั้นตอนที่ 1: ดึงเฉพาะข้อมูลที่ปฏิบัติงานจริง และค้นหาข้อมูลเก่าของผู้ใช้เดิมที่จะทำการเขียนทับ (Rule 3)
    for (var i = 1; i < existingRows.length; i++) {
      var row = existingRows[i];
      if (!row[COL.SWAP.DATE]) continue;

      var rowDateStr = parseRowDate(row[COL.SWAP.DATE]);
      var rowShift = String(row[COL.SWAP.SHIFT] || '').trim();
      var rowUserId = String(row[COL.SWAP.LINE_USER_ID] || '').trim();
      var rowSubId = String(row[colSubEmployeeId] || '').trim();

      var isSameDateAndShift = (dates.indexOf(rowDateStr) !== -1 && rowShift === proposedShift);

      if (isSameDateAndShift && rowUserId === reqLineUserId) {
        // บันทึก Index แถวเดิมของผู้ยื่นรายนี้ เพื่อเตรียมใช้ทำลายล้าง/เขียนทับใน Rule 3
        deleteIndices.push(i);
      } else {
        // เก็บรายชื่อและตารางงานของผู้ปฏิบัติหน้าที่แทนท่านอื่นที่ยังใช้งานอยู่
        if (rowSubId) {
          if (!activeSubstitutes[rowSubId]) {
            activeSubstitutes[rowSubId] = {};
          }
          activeSubstitutes[rowSubId][rowDateStr + '|' + rowShift] = true;
        }
      }
    }

    // ----------------------------------------------------
    // RULE 1 : ห้ามควงกะผลัดติดกัน (ตรวจทั้งผู้ลา และผู้แทน)
    // ----------------------------------------------------

    // 1. ตรวจสอบเงื่อนไขผู้แทนเวร (นำผลัดใหม่มาทดลองรวมกับตารางงานที่มีอยู่แล้วตรวจหาควงกะ)
    var subSchedule = {};
    if (activeSubstitutes[subEmpId]) {
      subSchedule = Object.assign({}, activeSubstitutes[subEmpId]);
    }
    dates.forEach(function(D) {
      subSchedule[D + '|' + proposedShift] = true;
    });

    if (hasConsecutiveShifts(subSchedule)) {
      logAction('submitSwap', payload.lineUserId, 'ลา', 'FAIL', 'ผู้แทนควงกะ');
      return jsonResponse(false, 'ห้ามควงกะผลัดติดกัน (ผู้แทนเวรปฏิบัติงานติดต่อกันเกินกำหนด)');
    }

    // 2. ตรวจสอบเงื่อนไขผู้ลา (เนื่องจากเป็นการลาออก จึงไม่นับกะใหม่เข้าไปในแผนการทำงาน)
    if (reqEmpId) {
      var reqSchedule = {};
      if (activeSubstitutes[reqEmpId]) {
        reqSchedule = Object.assign({}, activeSubstitutes[reqEmpId]);
      }
      if (hasConsecutiveShifts(reqSchedule)) {
        logAction('submitSwap', payload.lineUserId, 'ลา', 'FAIL', 'ผู้ลาควงกะ');
        return jsonResponse(false, 'ห้ามควงกะผลัดติดกัน (ผู้ลามีการปฏิบัติงานผลัดอื่นติดต่อกันเกินกำหนด)');
      }
    }

    // ----------------------------------------------------
    // RULE 2 : ผู้แทนซ้ำของคนอื่น
    // ----------------------------------------------------
    for (var di = 0; di < dates.length; di++) {
      var D = dates[di];
      if (activeSubstitutes[subEmpId] && activeSubstitutes[subEmpId][D + '|' + proposedShift]) {
        logAction('submitSwap', payload.lineUserId, 'ลา', 'FAIL', 'ผู้แทนซ้ำกับคนอื่น');
        return jsonResponse(false, 'ผู้แทนเวรคนนี้ถูกเลือกโดยเจ้าหน้าที่ท่านอื่นแล้ว');
      }
    }

    // ----------------------------------------------------
    // RULE 3 : ผู้แทนซ้ำของตัวเอง = แก้ไขข้อมูล
    // ----------------------------------------------------
    // ดำเนินการลบแถวเดิมที่บันทึกไว้ในวัน/เวลาเดียวกันออกก่อนเขียนชุดใหม่เข้าไปแทนที่ เพื่อรักษาความถูกต้องไม่ให้มีข้อมูลซ้ำซ้อน
    if (deleteIndices.length > 0) {
      deleteIndices.sort(function(a, b) { return b - a; });
      deleteIndices.forEach(function(idx) {
        sheet.deleteRow(idx + 1); // existingRows อ้างอิง index 0-based แต่ row ของ Google Sheets เป็น 1-based
      });
    }

    // 8. บันทึกข้อมูลใหม่เข้าสู่ตารางทำงาน
    var newRows = [];
    for (var di = 0; di < dates.length; di++) {
      newRows.push([
        dates[di],
        payload.lineUserId,
        auth.data.name || '',
        substituteName,
        payload.substituteEmployeeId,
        payload.shift,
        nowTimestamp()
      ]);
    }

    newRows.forEach(function(newRow) {
      sheet.appendRow(newRow);
    });

    // 9. จัดเรียงข้อมูลตามวันที่ (คอลัมน์ A)
    var lastRow = sheet.getLastRow();
    if (lastRow > 1) {
      sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).sort({column: 1, ascending: true});
    }

    logAction('submitSwap', payload.lineUserId, 'ลา', 'SUCCESS', dates.length + ' วัน, ผู้แทน: ' + substituteName);
    return jsonResponse(true, 'บันทึกการลาเรียบร้อย ✅ (' + dates.length + ' วัน)');
  } catch (err) {
    logAction('submitSwap', payload.lineUserId, 'ลา', 'ERROR', err.message);
    return handleError(err);
  } finally {
    lock.releaseLock();
  }
}

// ===== GET SWAP HISTORY (ประวัติการลา) =====

function getSwapHistory(payload) {
  try {
    var validation = validatePayload(payload, ['lineUserId']);
    if (!validation.valid) return jsonResponse(false, validation.message);

    var rows = SheetService.getValues(SHEETS.SWAP, 'A:G');
    var history = [];
    for (var i = rows.length - 1; i >= 1; i--) {
      if (rows[i][COL.SWAP.LINE_USER_ID] && String(rows[i][COL.SWAP.LINE_USER_ID]).trim() === payload.lineUserId.trim()) {
        history.push({
          date: rows[i][COL.SWAP.DATE] || '',
          name: rows[i][COL.SWAP.NAME] || '',
          substituteName: rows[i][COL.SWAP.SUBLIST_NAME] || '',
          shift: rows[i][COL.SWAP.SHIFT] || '',
          timestamp: rows[i][COL.SWAP.TIMESTAMP] || ''
        });
        if (history.length >= MAX_HISTORY) break;
      }
    }
    return jsonResponse(true, '', history);
  } catch (err) {
    return handleError(err);
  }
}
