// ===== GET SUBSTITUTE LIST (รายชื่อผู้แทนเวร) =====

function getSubstituteList() {
  try {
    var rows = SheetService.getValues(SHEETS.SUPLIST, 'A:C');
    if (rows.length < 2) return jsonResponse(true, 'ไม่มีข้อมูล', []);

    var list = [];
    for (var i = 1; i < rows.length; i++) {
      if (rows[i][COL.SUPLIST.NAME]) {
        list.push({
          name: rows[i][COL.SUPLIST.NAME] || '',
          employeeId: rows[i][COL.SUPLIST.EMPLOYEE_ID] || ''
        });
      }
    }

    // เรียงตามชื่อ ก-ฮ
    list.sort(function(a, b) {
      return a.name.localeCompare(b.name, 'th');
    });

    return jsonResponse(true, '', list);
  } catch (err) {
    return handleError(err);
  }
}

// ===== SUBMIT SWAP (บันทึกการลา) — รองรับลาห้วงเวลา =====

function submitSwap(payload) {
  var lock = LockService.getScriptLock();
  lock.waitLock(30000);
  try {
    // 1. Validate required fields
    var validation = validatePayload(payload, ['lineUserId', 'swapDateStart', 'swapDateEnd', 'shift', 'substituteEmployeeId']);
    if (!validation.valid) return jsonResponse(false, validation.message);

    // 2. Validate dates
    var dateStartValidation = validateDate(payload.swapDateStart);
    if (!dateStartValidation.valid) return jsonResponse(false, dateStartValidation.message);
    var dateEndValidation = validateDate(payload.swapDateEnd);
    if (!dateEndValidation.valid) return jsonResponse(false, dateEndValidation.message);

    if (payload.swapDateEnd < payload.swapDateStart) {
      return jsonResponse(false, 'วันที่สิ้นสุดต้องไม่ก่อนวันที่เริ่ม');
    }

    // 3. Authenticate (ตรวจสอบสิทธิ์) — ได้ชื่อผู้ลาและ employeeId
    var auth = authenticateRequest(payload);
    if (!auth.success) return jsonResponse(false, auth.message);

    // 4. ห้ามเลือกตัวเองเป็นผู้แทนเวร (เทียบ employeeId)
    if (auth.data.employeeId && auth.data.employeeId === payload.substituteEmployeeId) {
      return jsonResponse(false, 'ไม่สามารถเลือกตนเองเป็นผู้แทนเวรได้');
    }

    // 5. ดึงชื่อผู้แทน (จากชีท SUPLIST โดยใช้ employeeId)
    var substituteName = '';
    var listRows = SheetService.getValues(SHEETS.SUPLIST, 'A:C');
    for (var j = 1; j < listRows.length; j++) {
      if (listRows[j][COL.SUPLIST.EMPLOYEE_ID] && String(listRows[j][COL.SUPLIST.EMPLOYEE_ID]).trim() === payload.substituteEmployeeId) {
        substituteName = listRows[j][COL.SUPLIST.NAME] || '';
        break;
      }
    }

    // 6. สร้างรายการวันที่ทั้งหมดในช่วง
    var dates = [];
    var current = new Date(payload.swapDateStart);
    var end = new Date(payload.swapDateEnd);
    while (current <= end) {
      var y = current.getFullYear();
      var m = String(current.getMonth() + 1).padStart(2, '0');
      var d = String(current.getDate()).padStart(2, '0');
      dates.push(y + '-' + m + '-' + d);
      current.setDate(current.getDate() + 1);
    }

    // 7. หาแถวของตัวเองเก่า + แถวที่ substitute ถูกเลือกซ้ำ เพื่อลบ
    var sheet = SheetService.getSheet(SHEETS.SWAP);
    var existingRows = SheetService.getValues(SHEETS.SWAP, 'A:G');
    var requesterName = auth.data.name || '';
    var rowsToDelete = {};

    // เก็บ index แถวที่ต้องลบ (key = dateStr)
    for (var i = 1; i < existingRows.length; i++) {
      var row = existingRows[i];
      if (row[COL.SWAP.DATE] && row[COL.SWAP.LINE_USER_ID]) {
        var existingDate = String(row[COL.SWAP.DATE]).trim();
        var existingUserId = String(row[COL.SWAP.LINE_USER_ID]).trim();

        // ลบแถวเก่าของ user นี้ในวันที่ซ้ำกัน (upsert)
        if (existingUserId === payload.lineUserId) {
          for (var di = 0; di < dates.length; di++) {
            if (existingDate === dates[di]) {
              rowsToDelete[i] = true;
              break;
            }
          }
        }
      }
    }

    // ลบแถวเก่าทั้งหมด (เรียง index จากมากไปน้อย)
    var deleteIndices = [];
    for (var idx in rowsToDelete) {
      if (rowsToDelete.hasOwnProperty(idx)) {
        deleteIndices.push(parseInt(idx));
      }
    }
    deleteIndices.sort(function(a, b) { return b - a; });
    // แปลง index 0-based ของ array เป็น row number ใน sheet (+1 เพราะ header)
    for (var k = 0; k < deleteIndices.length; k++) {
      sheet.deleteRow(deleteIndices[k] + 1);
    }

    // 8. บันทึกข้อมูลทีละวัน
    for (var di = 0; di < dates.length; di++) {
      var values = [[
        dates[di],
        payload.lineUserId,
        requesterName,
        substituteName,
        payload.substituteEmployeeId,
        payload.shift,
        nowTimestamp()
      ]];
      SheetService.appendRow(SHEETS.SWAP, values[0]);
    }

    return jsonResponse(true, 'บันทึกการลาเรียบร้อย ✅ (' + dates.length + ' วัน)');
  } catch (err) {
    return handleError(err);
  } finally {
    lock.releaseLock();
  }
}

// ===== GET SWAP HISTORY (ประวัติการลา) =====

function getSwapHistory(payload) {
  try {
    var validation = validatePayload(payload, ['lineUserId']);
    if (!validation.valid) return jsonResponse(false, validation.message);

    var rows = SheetService.getValues(SHEETS.SWAP, 'A:G');
    var history = [];
    for (var i = rows.length - 1; i >= 1; i--) {
      if (rows[i][COL.SWAP.LINE_USER_ID] && String(rows[i][COL.SWAP.LINE_USER_ID]).trim() === payload.lineUserId.trim()) {
        history.push({
          date: rows[i][COL.SWAP.DATE] || '',
          name: rows[i][COL.SWAP.NAME] || '',
          substituteName: rows[i][COL.SWAP.SUBLIST_NAME] || '',
          shift: rows[i][COL.SWAP.SHIFT] || '',
          timestamp: rows[i][COL.SWAP.TIMESTAMP] || ''
        });
        if (history.length >= MAX_HISTORY) break;
      }
    }
    return jsonResponse(true, '', history);
  } catch (err) {
    return handleError(err);
  }
}
